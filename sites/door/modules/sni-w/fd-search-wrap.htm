/* allows label to lay on top of search bar, but disappear when there is content or user focuses on search bar. */ (function($) { $(function() { $('.fd-search') .addClass('ready') .find('form') .each(function() { $form = $(this); $label = $form.find('p.input label'); $input = $form.find('p.input input'); $label.hide(); $input .attr('title', $label.text()) .addClass('ready') .focus(function() { $label.hide(); }) .blur(function() { if($input.val() == '') $label.show(); }) .blur(); }); }); });
